  73.   public function testQueryChains() {
  74.     $sql = 'SELECT * FROM qpdb_test';
  75.     $args = array();
  76.     $qp = qp(QueryPath::HTML_STUB, 'body') // Open a stub HTML doc and select <body/>
  77.       ->append('<h1></h1>') // Add <h1/>
  78.       ->children()  // Select the <h1/>
  79.       //->dbInit($this->dsn) // Connect to the database
  80.       ->query($sql, $args) // Execute the SQL query
  81.       ->nextRow()  // Select a row. By default, no row is selected.
  82.       //->withEachRow()
  83.       ->appendColumn('colOne') // Append Row 1, Col 1 (Title 0)
  84.       ->parent() // Go back to the <body/>
  85.       ->append('<p/>') // Append a <p/> to the body
  86.       ->find('p')  // Find the <p/> we just created.
  87.       ->nextRow() // Advance to row 2
  88.       ->prependColumn('colTwo') // Get row 2, col 2. (Body 1)
  89.       ->columnAfter('colThree') // Get row 2 col 3. (Footer 1)
  90.       ->doneWithQuery() // Let QueryPath clean up.
  91.       ;//->writeHTML(); // Write the output as HTML.
  92.     $this->assertEquals('Title 0', $qp->top()->find('h1')->text());
  93.     $this->assertEquals('Body 1', $qp->top()->find('p')->text());
  94.     
  95.     $qp = qp(QueryPath::HTML_STUB, 'body') // Open a stub HTML doc and select <body/>
  96.       ->append('<table><tbody/></table>')
  97.       ->find('tbody')
  98.       ->query('SELECT * FROM qpdb_test LIMIT 2')
  99.       ->withEachRow()
  100.       ->appendColumn('colOne')
  101.       ->doneWithQuery();
  102.     $this->assertEquals('Title 0Title 1', $qp->top()->find('tbody')->text());
  103.     
  104.     $wrap = '<?xml version="1.0"?><tr><td/></tr>';
