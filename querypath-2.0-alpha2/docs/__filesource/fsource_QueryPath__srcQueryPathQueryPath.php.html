  137.  *     their decimal equivalent, and will replace un-escaped ampersands with
  138.  *     a numeric entity equivalent.

  140.  * @example examples/simple_example.php Basic Example

  360.     
  361.     // Optimize for ID/Class searches. These two take a long time
  362.     // when a rdp is used. Using an XPath pushes work to C code.
  363.     $ids = array();
  364.     $regex = '/^#([\w-]+)$|^\.([\w-]+)$/'; // $1 is ID, $2 is class.
  365.     //$regex = '/^#([\w-]+)$/';
  366.     if (preg_match($regex, $selector, $ids) === 1) {
  367.       // If $1 is a match, we have an ID.
  368.       if (!empty($ids[1])) {
  369.         $xpath = new DOMXPath($this->document);
  370.         foreach ($this->matches as $item) {
  371.           $nl = $xpath->query("//*[@id='{$ids[1]}']", $item);
  372.           if ($nl->length > 0) {
  373.             $this->setMatches($nl->item(0));
  374.             break;
  375.           }
  376.           else {
  377.             // If no match is found, we set an empty.
  378.             $this->noMatches();
  379.           }
  380.         }
  381.       }
  382.       // Quick search for class values. While the XPath can't do it
  383.       // all, it is faster than doing a recusive node search.
  384.       else {
  385.         //$this->xpath("//*[@class='{$ids[2]}']");
  386.         $xpath = new DOMXPath($this->document);
  387.         $found = new SplObjectStorage();

  451.   /**

  453.    *

  455.    * the QueryPath. Typically, these are DOMElement objects (unless you have
  456.    * used {@link map()}, {@link xpath()}, or other methods that can select
  457.    * non-elements).

  459.    * When called with an index, it will return the item in the QueryPath with
  460.    * that index number.

  466.    * also iterate over qp() itself (QueryPath implementations must be Traversable).
  467.    * In the later case, though, each item
  468.    * will be wrapped in a QueryPath object. To learn more about iterating
  469.    * in QueryPath, see {@link examples/techniques.php}.
  470.    *
  471.    * @param int $index

  476.    *    instead of an array. This is the preferred method for extensions to use.
  477.    * @return mixed

  479.    *    present, an array of all matches will be returned.

  488.     if (!$asObject) {

  495.   /**
  496.    * Get/set an attribute.
  497.    * - If both name and value are set, then this will set the attribute name/value
  498.    *   pair for all items in this object.
  499.    * - If name is set, and is an array, then
  500.    *   all attributes in the array will be set for all items in this object.
  501.    * - If name is a string and is set, then the attribute value will be returned.
  502.    *
  503.    * When an attribute value is retrieved, only the attribute value of the FIRST
  504.    * match is returned.
  505.    *
  506.    * @param mixed $name 
  507.    *    The name of the attribute or an associative array of name/value pairs.
  508.    * @param string $value 
  509.    *    A value (used only when setting an individual property).
  510.    * @return mixed 
  511.    *    If this was a setter request, return the QueryPath object. If this was
  512.    *    an access request (getter), return the string value.
  513.    * @see removeAttr()
  514.    * @see tag()
  515.    */
  516.   public function attr($name, $value = NULL) {
  517.     // multi-setter
  518.     if (is_array($name)) {
  519.       foreach ($name as $k => $v) {
  520.         foreach ($this->matches as $m) $m->setAttribute($k, $v);
  521.       }
  522.       return $this;
  523.     }
  524.     // setter
  525.     if (isset($value)) {
  526.       foreach ($this->matches as $m) $m->setAttribute($name, $value);
  527.       return $this;
  528.     }
  529.     
  530.     //getter
  531.     if ($this->matches->count() == 0) return NULL;
  532.     
  533.     // Special node type handler:
  534.     if ($name == 'nodeType') {
  535.       return $this->getFirstMatch()->nodeType;
  536.     }
  537.     
  538.     // Always return first match's attr.
  539.     return $this->getFirstMatch()->getAttribute($name);
  540.   }
  541.   /**
  542.    * Set/get a CSS value for the current element(s).
  543.    * This sets the CSS value for each element in the QueryPath object.
  544.    * It does this by setting (or getting) the style attribute (without a namespace).
  545.    *
  546.    * For example, consider this code:
  547.    * <code>
  548.    * <?php
  549.    * qp(HTML_STUB, 'body')->css('background-color','red')->html();
  550.    * ?>
  551.    * </code>
  552.    * This will return the following HTML:
  553.    * <code>
  554.    * <body style="background-color: red"/>
  555.    * </code>
  556.    *
  557.    * If no parameters are passed into this function, then the current style
  558.    * element will be returned unparsed. Example:
  559.    * <code>
  560.    * <?php
  561.    * qp(HTML_STUB, 'body')->css('background-color','red')->css();
  562.    * ?>
  563.    * </code>
  564.    * This will return the following:
  565.    * <code>
  566.    * background-color: red
  567.    * </code>
  568.    *
  569.    * @param mixed $name 
  570.    *   If this is a string, it will be used as a CSS name. If it is an array,
  571.    *   this will assume it is an array of name/value pairs of CSS rules. It will
  572.    *   apply all rules to all elements in the set.
  573.    * @return QueryPath 
  574.    */
  575.   public function css($name = NULL, $value = '') {
  576.     if (empty($name)) {
  577.       return $this->attr('style');
  578.     }
  579.     $format = '%s: %s';
  580.     if (is_array($name)) {
  581.       $buf = array();
  582.       foreach ($name as $key => $val) {
  583.         $buf[] = sprintf($format, $key, $val);
  584.       }
  585.       implode(';', $buf);
  586.     }
  587.     else {
  588.       $css = sprintf($format, $name, $value);
  589.     }
  590.     $this->attr('style', $css);
  591.     return $this;
  592.   }

  676.   /**
  677.    * Filter based on a lambda function.
  678.    *
  679.    * The function string will be executed as if it were the body of a
  680.    * function. It is passed two arguments:
  681.    * - $index: The index of the item.
  682.    * - $item: The current Element.
  683.    * If the function returns boolean FALSE, the item will be removed from
  684.    * the list of elements. Otherwise it will be kept.
  685.    *
  686.    * Example:
  687.    * <code>
  688.    * qp('li')->filterLambda('qp($item)->attr("id") == "test"');
  689.    * </code>
  690.    *
  691.    * The above would filter down the list to only an item whose ID is
  692.    * 'text'.
  693.    *
  694.    * @param string $function 
  695.    *   Inline lambda function in a string.
  696.    * @return QueryPath 
  697.    * @see filter()
  698.    * @see map()
  699.    * @see mapLambda()
  700.    * @see filterCallback()
  701.    */
  702.   public function filterLambda($fn) {
  703.     $function = create_function('$index, $item', $fn);
  704.     $found = new SplObjectStorage();
  705.     $i = 0;
  706.     foreach ($this->matches as $item)
  707.       if ($function($i++, $item) !== FALSE) $found->attach($item);
  708.     
  709.     $this->setMatches($found);
  710.     return $this;
  711.   }
  712.   /**
  713.    * Filter based on a callback function.
  714.    *
  715.    * A callback may be any of the following:
  716.    *  - a function: 'my_func'.
  717.    *  - an object/method combo: $obj, 'myMethod'
  718.    *  - a class/method combo: 'MyClass', 'myMethod'
  719.    * Note that classes are passed in strings. Objects are not.
  720.    *
  721.    * Each callback is passed to arguments:
  722.    *  - $index: The index position of the object in the array.
  723.    *  - $item: The item to be operated upon.
  724.    *
  725.    * @param $callback. 
  726.    *    A callback either as a string (function) or an array (object, method OR
  727.    *    classname, method).
  728.    * @return QueryPath 
  729.    *    Query path object augmented according to the function.
  730.    * @see filter()
  731.    * @see filterLambda()
  732.    * @see map()
  733.    * @see is()
  734.    * @see find()
  735.    */
  736.   public function filterCallback($callback) {
  737.     $found = new SplObjectStorage();
  738.     $i = 0;
  739.     if (is_callable($callback)) {
  740.       foreach($this->matches as $item) 
  741.         if (call_user_func($callback, $i++, $item) !== FALSE) $found->attach($item);
  742.     }
  743.     else {
  744.       throw new QueryPathException('The specified callback is not callable.');
  745.     }
  746.     $this->setMatches($found);
  747.     return $this;
  748.   }
  749.   /**
  750.    * Filter a list to contain only items that do NOT match.
  751.    *
  752.    * @param string $selector 
  753.    *   A selector to use as a negation filter. If the filter is matched, the
  754.    *   element will be removed from the list.
  755.    * @return QueryPath 
  756.    *   The QueryPath object with matching items filtered out.
  757.    * @see find()
  758.    */
  759.   public function not($selector) {
  760.     $found = new SplObjectStorage();
  761.     if ($selector instanceof DOMElement) {
  762.       foreach ($this->matches as $m) if ($m !== $selector) $found->attach($m); 
  763.     }
  764.     elseif (is_array($selector)) {
  765.       foreach ($this->matches as $m) if (!in_array($m, $selector)) $found->attach($m); 
  766.     }
  767.     elseif ($selector instanceof SplObjectStorage) {
  768.       foreach ($this->matches as $m) if ($selector->contains($m)) $found->attach($m); 
  769.     }
  770.     else {
  771.       foreach ($this->matches as $m) if (!qp($m)->is($selector)) $found->attach($m);
  772.     }
  773.     $this->setMatches($found);
  774.     return $this;
  775.   }
  776.   /**
  777.    * Get an item's index.
  778.    *
  779.    * Given a DOMElement, get the index from the matches. This is the
  780.    * converse of {@link get()}.
  781.    *
  782.    * @param DOMElement $subject 
  783.    *   The item to match.
  784.    * 
  785.    * @return mixed 
  786.    *   The index as an integer (if found), or boolean FALSE. Since 0 is a
  787.    *   valid index, you should use strong equality (===) to test..
  788.    * @see get()
  789.    * @see is()
  790.    */
  791.   public function index($subject) {
  792.     
  793.     $i = 0;
  794.     foreach ($this->matches as $m) {
  795.       if ($m === $subject) return $i;
  796.       ++$i;
  797.     }
  798.     return FALSE;
  799.   }
  800.   /**
  801.    * Run a function on each item in a set.
  802.    *
  803.    * The mapping callback can return anything. Whatever it returns will be
  804.    * stored as a match in the set, though. This means that afer a map call,
  805.    * there is no guarantee that the elements in the set will behave correctly
  806.    * with other QueryPath functions.
  807.    *
  808.    * Callback rules:
  809.    * - If the callback returns NULL, the item will be removed from the array.
  810.    * - If the callback returns an array, the entire array will be stored in
  811.    *   the results.
  812.    * - If the callback returns anything else, it will be appended to the array
  813.    *   of matches.
  814.    *
  815.    * @param callback $callback

  898.   /**

  903.    *   special treatment of $this, as there is in jQuery.)

  909.    * - All other return values are ignored.
  910.    *
  911.    * @param callback $callback 
  912.    *   The callback to run.

  914.    *   The QueryPath.

  916.    * @see filter()

  922.       foreach ($this->matches as $item) {
  923.         if (call_user_func($callback, $i, $item) === FALSE) return $this;
  924.         ++$i;
  925.       }
  926.     }
  927.     else {
  928.       throw new Exception('Callback is not callable.');
  929.     }

  931.   }

  934.    * 
  935.    * @param string $lambda 
  936.    *   The lambda function. This will be passed ($index, &$item).
  937.    * @return QueryPath 
  938.    *   The QueryPath object.

  952.   }
  953.   /**
  954.    * Insert the given markup as the last child.
  955.    *
  956.    * The markup will be inserted into each match in the set.
  957.    *
  958.    * The same element cannot be inserted multiple times into a document. DOM
  959.    * documents do not allow a single object to be inserted multiple times
  960.    * into the DOM. To insert the same XML repeatedly, we must first clone
  961.    * the object. This has one practical implication: Once you have inserted
  962.    * an element into the object, you cannot further manipulate the original
  963.    * element and expect the changes to be replciated in the appended object.
  964.    * (They are not the same -- there is no shared reference.) Instead, you
  965.    * will need to retrieve the appended object and operate on that.
  966.    *
  967.    * @param mixed $data 
  968.    *   This can be either a string (the usual case), or a DOM Element.
  969.    * @return QueryPath 
  970.    *   The QueryPath object.
  971.    * @see appendTo()
  972.    * @see prepend()
  973.    */
  974.   public function append($data) {
  975.     $data = $this->prepareInsert($data);
  976.     if (isset($data)) {
  977.       if (empty($this->document->documentElement) && $this->matches->count() == 0) {
  978.         // Then we assume we are writing to the doc root
  979.         $this->document->appendChild($data);
  980.         $found = new SplObjectStorage();
  981.         $found->attach($this->document->documentElement);
  982.         $this->setMatches($found);
  983.       }
  984.       else {
  985.         // You can only append in item once. So in cases where we
  986.         // need to append multiple times, we have to clone the node.
  987.         foreach ($this->matches as $m) { 
  988.           // DOMDocumentFragments are even more troublesome, as they don't
  989.           // always clone correctly. So we have to clone their children.
  990.           if ($data instanceof DOMDocumentFragment) {
  991.             foreach ($data->childNodes as $n)
  992.               $m->appendChild($n->cloneNode(TRUE));
  993.           }
  994.           else {
  995.             // Otherwise a standard clone will do.
  996.             $m->appendChild($data->cloneNode(TRUE));
  997.           }
  998.           
  999.         }
  1000.       }
  1001.         
  1002.     }
  1003.     return $this;
  1004.   }
