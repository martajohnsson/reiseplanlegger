  16. require_once '../src/QueryPath/QueryPath.php';
  17.  
  18. $artist_url = 'http://musicbrainz.org/ws/1/artist/?type=xml&name=u2';
  19. $album_url = 'http://musicbrainz.org/ws/1/release/?type=xml&artistid=';
  20. try {
  21.   $artist = qp($artist_url, 'artist:first');

  29.

  32.       // Fixme: Label is broken. See Drupal QueryPath module.
  33.       print '(' . $album->next('label')->text() . ')' .PHP_EOL;
  34.     }
  35.   }
  36. }
  37. catch (Exception $e) {
