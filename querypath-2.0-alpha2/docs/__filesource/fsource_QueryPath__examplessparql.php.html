  2. /**
  3.  * Use QueryPath to query semantic servers.
  4.  *
  5.  * This demo shows how a more complex GET query can be built up in
  6.  * QueryPath. POST queries are supported, too. Use a stream context
  7.  * to create those.
  8.  */
  9. require '../src/QueryPath/QueryPath.php';
  10.  
  11. // We are using the dbpedia database to execute a SPARQL query.
  12.  
  13. // URL to DB Pedia's SPARQL endpoint.
  14. $url = 'http://dbpedia.org/sparql';
  15.  
  16. // The SPARQL query to run.
  17. $sparql = '
  18.   PREFIX foaf: <http://xmlns.com/foaf/0.1/>
  19.   PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
  20.   SELECT ?uri ?name ?label
  21.   WHERE {
  22.     ?uri foaf:name ?name .
  23.     ?uri rdfs:label ?label
  24.     FILTER (?name = "The Beatles")
  25.     FILTER (lang(?label) = "en")
  26.   }
  27. ';
  28.  
  29. // We first set up the parameters that will be sent.
  30. $params = array(
  31.   'query' => $sparql,
  32.   'format' => 'application/sparql-results+xml',
  33. );
  34.  
  35. // DB Pedia wants a GET query, so we create one.
  36. $data = http_build_query($params);
  37. $url .= '?' . $data;
  38.  
  39. // Next, we simply retrieve, parse, and output the contents.
  40. $qp = qp($url, 'head');
  41.  
  42. // Get the headers from the resulting XML.
  43. $headers = array();
  44. foreach ($qp->children('variable') as $col) {
  45.   $headers[] = $col->attr('name');
  46. }
  47.  
  48. // Get rows of data from result.
  49. $rows = array();
  50. $col_count = count($headers);
  51. foreach ($qp->top()->find('results>result') as $row) {
  52.   $cols = array();
  53.   $row->children();
  54.   for ($i = 0; $i < $col_count; ++$i) {
  55.     $cols[$i] = $row->branch()->eq($i)->text();
  56.   }
  57.   $rows[] = $cols;
  58. }
  59.  
  60. // Turn data into table.
  61. $table = '<table><tr><th>' . implode('</th><th>', $headers) . '</th></tr>';
  62. foreach ($rows as $row) {
  63.   $table .= '<tr><td>';
  64.   $table .= implode('</td><td>', $row);
  65.   $table .= '</td></tr>';
  66. }
  67. $table .= '</table>';
  68.  
  69. // Add table to HTML document.
  70. qp(QueryPath::HTML_STUB, 'body')->append($table)->writeHTML();
