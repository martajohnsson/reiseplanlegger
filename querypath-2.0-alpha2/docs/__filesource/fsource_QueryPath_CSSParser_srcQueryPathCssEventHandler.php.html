  2. /**
  3.  * This file contains a full implementation of the CssEventHandler interface.
  4.  * 
  5.  * The tools in this package initiate a CSS selector parsing routine and then
  6.  * handle all of the callbacks.
  7.  *
  8.  * The implementation provided herein adheres to the CSS 3 Selector specification
  9.  * with the following caveats:
  10.  *
  11.  *  - The negation (:not()) and containment (:has()) pseudo-classes allow *full*
  12.  *    selectors and not just simple selectors.
  13.  *  - There are a variety of additional pseudo-classes supported by this
  14.  *    implementation that are not part of the spec. Most of the jQuery
  15.  *    pseudo-classes are supported. The :x-root pseudo-class is also supported.
  16.  *  - Pseudo-classes that require a User Agent to function have been disabled.
  17.  *    Thus there is no :hover pseudo-class.
  18.  *  - All pseudo-elements require the double-colon (::) notation. This breaks
  19.  *    backward compatibility with the 2.1 spec, but it makes visible the issue
  20.  *    that pseudo-elements cannot be effectively used with most of the present
  21.  *    library. They return <b>stdClass objects with a text property</b> (QP > 1.3)
  22.  *    instead of elements.
  23.  *  - The pseudo-classes first-of-type, nth-of-type and last-of-type may or may
  24.  *    not conform to the specification. The spec is unclear.
  25.  *  - pseudo-class filters of the form -an+b do not function as described in the
  26.  *    specification. However, they do behave the same way here as they do in
  27.  *    jQuery.
  28.  *  - This library DOES provide XML namespace aware tools. Selectors can use
  29.  *    namespaces to increase specificity.
  30.  *  - This library does nothing with the CSS 3 Selector specificity rating. Of
  31.  *    course specificity is preserved (to the best of our abilities), but there
  32.  *    is no calculation done.
  33.  *
  34.  * For detailed examples of how the code works and what selectors are supported,
  35.  * see the {@see CssEventTests.php} file, which contains the unit tests used for
  36.  * testing this implementation.
  37.  *
  38.  * @package QueryPath
  39.  * @subpackage CSSParser
  40.  * @author M Butcher <matt@aleph-null.tv>
  41.  * @license http://opensource.org/licenses/lgpl-2.1.php LGPL (The GNU Lesser GPL) or an MIT-like license.
  42.  */

  49. /**
  50.  * Handler that tracks progress of a query through a DOM.
  51.  *
  52.  * The main idea is that we keep a copy of the tree, and then use an
  53.  * array to keep track of matches. To handle a list of selectors (using
  54.  * the comma separator), we have to track both the currently progressing
  55.  * match and the previously matched elements.
  56.  *
  57.  * To use this handler:
  58.  * <code>
  59.  * $filter = '#id'; // Some CSS selector
  60.  * $handler = new QueryPathCssParser(DOMNode $dom);
  61.  * $parser = new CssParser();
  62.  * $parser->parse($filter, $handler);
  63.  * $matches = $handler->getMatches();
  64.  * </code>
  65.  *
  66.  * $matches will be an array of zero or more DOMElement objects.

  366.   /**
  367.    * This also supports the following nonstandard pseudo classes:
  368.    *  - :x-reset/:x-root (reset to the main item passed into the constructor. Less drastic than :root)
  369.    *  - :odd/:even (shorthand for :nth-child(odd)/:nth-child(even))
  370.    */
  371.   public function pseudoClass($name, $value = NULL) {
  372.     $name = strtolower($name);
  373.     // Need to handle known pseudoclasses.
  374.     switch($name) {
  375.       case 'visited':
  376.       case 'hover':
  377.       case 'active':
  378.       case 'focus':
  379.       case 'animated': //  Last 3 are from jQuery
  380.       case 'visible':
  381.       case 'hidden':
  382.         // These require a UA, which we don't have.
  383.       case 'target':
  384.         // This requires a location URL, which we don't have.
  385.         $this->matches = new SplObjectStorage();
  386.         break;
  387.       case 'indeterminate':
  388.         // The assumption is that there is a UA and the format is HTML.
  389.         // I don't know if this should is useful without a UA.
  390.         throw new NotImplementedException(":indeterminate is not implemented.");
  391.         break;
  392.       case 'lang':
  393.         // No value = exception.
  394.         if (!isset($value)) {
  395.           throw new NotImplementedException("No handler for lang pseudoclass without value.");
  396.         }
  397.         $this->lang($value);
  398.         break;
  399.       case 'link':
  400.         $this->searchForAttr('href');
  401.         break;
  402.       case 'root':
  403.         $found = new SplObjectStorage();
  404.         if (empty($this->dom)) {
  405.           $this->matches = $found;
  406.         }
  407.         elseif (is_array($this->dom)) {
  408.           $found->attach($this->dom[0]->ownerDocument->documentElement);
  409.           $this->matches = $found;
  410.         }
  411.         elseif ($this->dom instanceof DOMNode) {
  412.           $found->attach($this->dom->ownerDocument->documentElement);
  413.           $this->matches = $found;
  414.         }
  415.         elseif ($this->dom instanceof DOMNodeList && $this->dom->length > 0) {
  416.           $found->attach($this->dom->item(0)->ownerDocument->documentElement);
  417.           $this->matches = $found;
  418.         }
  419.         else {
  420.           // Hopefully we never get here:
  421.           $found->attach($this->dom);
  422.           $this->matches = $found;
  423.         }
  424.         break;
  425.       
  426.       // NON-STANDARD extensions for reseting to the "top" items set in
  427.       // the constructor.  
  428.       case 'x-root':
  429.       case 'x-reset':
  430.         $this->matches = new SplObjectStorage();
  431.         $this->matches->attach($this->dom);
  432.         break;        
  433.       
  434.       // NON-STANDARD extensions for simple support of even and odd. These
  435.       // are supported by jQuery, FF, and other user agents.  
  436.       case 'even':
  437.         $this->nthChild(2, 0);
  438.         break;
  439.       case 'odd':
  440.         $this->nthChild(2, 1);
  441.         break;
  442.       
  443.       // Standard child-checking items.  
  444.       case 'nth-child':
  445.         list($aVal, $bVal) = $this->parseAnB($value);
  446.         $this->nthChild($aVal, $bVal);
  447.         break;
  448.       case 'nth-last-child':
  449.         list($aVal, $bVal) = $this->parseAnB($value);
  450.         $this->nthLastChild($aVal, $bVal);
  451.         break;
  452.       case 'nth-of-type':
  453.         list($aVal, $bVal) = $this->parseAnB($value);
  454.         $this->nthOfTypeChild($aVal, $bVal);
  455.         break;
  456.       case 'nth-last-of-type':
  457.         list($aVal, $bVal) = $this->parseAnB($value);
  458.         $this->nthLastOfTypeChild($aVal, $bVal);
  459.         break;
  460.       case 'first-child':
  461.         $this->nthChild(0, 1);
  462.         break;
  463.       case 'last-child':
  464.         $this->nthLastChild(0, 1);
  465.         break;
  466.       case 'first-of-type':
  467.         $this->firstOfType();
  468.         break;
  469.       case 'last-of-type':
  470.         $this->lastOfType();
  471.         break;
  472.       case 'only-child':
  473.         $this->onlyChild();
  474.         break;
  475.       case 'only-of-type':
  476.         $this->onlyOfType();
  477.         break;
  478.       case 'empty':
  479.         $this->emptyElement();
  480.         break;  
  481.       case 'not':
  482.         if (empty($value)) {
  483.           throw new CssParseException(":not() requires a value.");
  484.         }
  485.         $this->not($value);
  486.         break;
  487.       // Additional pseudo-classes defined in jQuery:
  488.       case 'lt':
  489.       case 'gt':
  490.       case 'nth':
  491.       case 'eq':
  492.       case 'first':
  493.       case 'last':
  494.       //case 'even':
  495.       //case 'odd':
  496.         $this->getByPosition($name, $value);  
  497.         break;
  498.       case 'parent':
  499.         $matches = $this->candidateList();
  500.         $found = new SplObjectStorage();
  501.         foreach ($matches as $match) {
  502.           if (!empty($match->firstChild)) {
  503.             $found->attach($match);
  504.           }
  505.         }
  506.         $this->matches = $found;
  507.         break;
  508.       
  509.       case 'enabled':  
  510.       case 'disabled':  
  511.       case 'checked':  
  512.         $this->attribute($name);
  513.         break;
  514.       case 'text':
  515.       case 'radio':
  516.       case 'checkbox':
  517.       case 'file':
  518.       case 'password':
  519.       case 'submit':
  520.       case 'image':
  521.       case 'reset':
  522.       case 'button':
  523.       case 'submit':
  524.         $this->attribute('type', $name);
  525.         break;
  526.  
  527.       case 'header':
  528.         $matches = $this->candidateList();
  529.         $found = new SplObjectStorage();
  530.         foreach ($matches as $item) {
  531.           $tag = $item->tagName;
  532.           $f = strtolower(substr($tag, 0, 1));
  533.           if ($f == 'h' && strlen($tag) == 2 && ctype_digit(substr($tag, 1, 1))) {
  534.             $found->attach($item);
  535.           }
  536.         }
  537.         $this->matches = $found;
  538.         break;
  539.       case 'has':
  540.         $this->has($value);
  541.         break;
  542.       // Contains == text matches.
  543.       case 'contains':
  544.         $matches = $this->candidateList();
  545.         $found = new SplObjectStorage();
  546.         foreach ($matches as $item) {
  547.           if ($item->textContent == $value) {
  548.             $found->attach($item);
  549.           }
  550.         }
  551.         $this->matches = $found;
  552.         break;
  553.       default:
  554.         throw new CssParseException("Unknown Pseudo-Class: " . $name);
  555.     }
  556.     $this->findAnyElement = FALSE;
  557.   }

  559.   /**
  560.    * Pseudo-class handler for a variety of jQuery pseudo-classes.
  561.    * Handles lt, gt, eq, nth, first, last pseudo-classes.
  562.    */
  563.   private function getByPosition($operator, $pos) {
  564.     $matches = $this->candidateList();
  565.     $found = new SplObjectStorage();
  566.     if ($matches->count() == 0) {
  567.       return;
  568.     }
  569.     
  570.     switch ($operator) {
  571.       case 'nth':
  572.       case 'eq':
  573.         if ($matches->count() >= $pos) {
  574.           //$found[] = $matches[$pos -1];
  575.           foreach ($matches as $match) {
  576.             // CSS is 1-based, so we pre-increment.
  577.             if ($matches->key() + 1 == $pos) {
  578.               $found->attach($match);
  579.               break;
  580.             }
  581.           }
  582.         }
  583.         break;
  584.       case 'first':
  585.         if ($matches->count() > 0) {
  586.           $matches->rewind(); // This is necessary to init.
  587.           $found->attach($matches->current());
  588.         }
  589.         break;
  590.       case 'last':
  591.         if ($matches->count() > 0) {
  592.           
  593.           // Spin through iterator.
  594.           foreach ($matches as $item) {};
  595.          
  596.           $found->attach($item);
  597.         }
  598.         break;
  599.       // case 'even': 
  600.       //         for ($i = 1; $i <= count($matches); ++$i) {
  601.       //           if ($i % 2 == 0) {
  602.       //             $found[] = $matches[$i];
  603.       //           }
  604.       //         }
  605.       //         break;
  606.       //       case 'odd':
  607.       //         for ($i = 1; $i <= count($matches); ++$i) {
  608.       //           if ($i % 2 == 0) {
  609.       //             $found[] = $matches[$i];
  610.       //           }
  611.       //         }
  612.       //         break;
  613.       case 'lt':
  614.         $i = 0;
  615.         foreach ($matches as $item) {
  616.           if (++$i < $pos) {
  617.             $found->attach($item);
  618.           }
  619.         }
  620.         break;
  621.       case 'gt':
  622.         $i = 0;
  623.         foreach ($matches as $item) {
  624.           if (++$i > $pos) {
  625.             $found->attach($item);
  626.           }
  627.         }
  628.         break;
  629.     }
  630.     
  631.     $this->matches = $found;
  632.   }

  634.   /**
  635.    * Parse an an+b rule for CSS pseudo-classes.
  636.    * @param $rule 
  637.    *   Some rule in the an+b format.
  638.    * @return 
  639.    *   Array (list($aVal, $bVal)) of the two values.
  640.    * @throws CssParseException
  641.    *   If the rule does not follow conventions.
  642.    */
  643.   protected function parseAnB($rule) {
  644.     if ($rule == 'even') {
  645.       return array(2, 0);
  646.     }
  647.     elseif ($rule == 'odd') {
  648.       return array(2, 1);
  649.     }
  650.     elseif ($rule == 'n') {
  651.       return array(1, 0);
  652.     }
  653.     elseif (is_numeric($rule)) {
  654.       return array(0, (int)$rule);
  655.     }
  656.     
  657.     $rule = explode('n', $rule);
  658.     if (count($rule) == 0) {
  659.       throw new CssParseException("nth-child value is invalid.");
  660.     }
  661.     $aVal = (int)trim($rule[0]);
  662.     $bVal = !empty($rule[1]) ? (int)trim($rule[1]) : 0;
  663.     return array($aVal, $bVal);
  664.   }
  665.   
  666.   /**
  667.    * Pseudo-class handler for nth-child and all related pseudo-classes.
  668.    *
  669.    * @param int $groupSize 
  670.    *   The size of the group (in an+b, this is a).
  671.    * @param int $elementInGroup 
  672.    *   The offset in a group. (in an+b this is b).
  673.    * @param boolean $lastChild 
  674.    *   Whether counting should begin with the last child. By default, this is false.
  675.    *   Pseudo-classes that start with the last-child can set this to true.
  676.    */
  677.   protected function nthChild($groupSize, $elementInGroup, $lastChild = FALSE) {
  678.     // EXPERIMENTAL: New in Quark. This should be substantially faster 
  679.     // than the old (jQuery-ish) version. It still has E_STRICT violations
  680.     // though.
  681.     $parents = new SplObjectStorage();
  682.     $matches = new SplObjectStorage();
  683.     
  684.     $i = 0;
  685.     foreach ($this->matches as $item) {
  686.       $parent = $item->parentNode;
  687.       
  688.       // Build up an array of all of children of this parent, and store the 
  689.       // index of each element for reference later. We only need to do this 
  690.       // once per parent, though.
  691.       if (!$parents->contains($parent)) {
  692.         
  693.         $c = 0;
  694.         foreach ($parent->childNodes as $child) {
  695.           // We only want nodes, and if this call is preceded by an element 
  696.           // selector, we only want to match elements with the same tag name.
  697.           // !!! This last part is a grey area in the CSS 3 Selector spec. It seems
  698.           // necessary to make the implementation match the examples in the spec. However,
  699.           // jQuery 1.2 does not do this.
  700.           if ($child->nodeType == XML_ELEMENT_NODE && ($this->findAnyElement || $child->tagName == $item->tagName)) {
  701.             // This may break E_STRICT.
  702.             $child->nodeIndex = ++$c;
  703.           }
  704.         }
  705.         // This may break E_STRICT.
  706.         $parent->numElements = $c;
  707.         $parents->attach($parent);
  708.       }
  709.       
  710.       // If we are looking for the last child, we count from the end of a list.
  711.       // Note that we add 1 because CSS indices begin at 1, not 0.
  712.       if ($lastChild) {
  713.         $indexToMatch = $item->parentNode->numElements  - $item->nodeIndex + 1;
  714.       }
  715.       // Otherwise we count from the beginning of the list.
  716.       else {
  717.         $indexToMatch = $item->nodeIndex;
  718.       }
  719.       
  720.       // If group size is 0, then we return element at the right index.
  721.       if ($groupSize == 0) {
  722.         if ($indexToMatch == $elementInGroup) 
  723.           $matches->attach($item);
  724.       }
  725.       // If group size != 0, then we grab nth element from group offset by
  726.       // element in group.
  727.       else {
  728.         if (($indexToMatch - $elementInGroup) % $groupSize == 0 
  729.             && ($indexToMatch - $elementInGroup) / $groupSize >= 0) {
  730.           $matches->attach($item);
  731.         }
  732.       }
  733.       
  734.       // Iterate.
  735.       ++$i;
  736.     }
  737.     $this->matches = $matches;
  738.   }
  739.   
  740.   /**
  741.    * Reverse a set of matches.
  742.    *
  743.    * This is now necessary because internal matches are no longer represented
  744.    * as arrays.
  745.    * @since QueryPath 2.0
  746.    */
  747.   private function reverseMatches() {
  748.     // Reverse the candidate list. There must be a better way of doing
  749.     // this.
  750.     $arr = array();
  751.     foreach ($this->matches as $m) array_unshift($arr, $m);
  752.     
  753.     $this->found = new SplObjectStorage();
  754.     foreach ($arr as $item) $this->found->attach($item);
  755.   }
  756.   
  757.   /**
  758.    * Pseudo-class handler for :nth-last-child and related pseudo-classes.
  759.    */
  760.   protected function nthLastChild($groupSize, $elementInGroup) {
  761.     // New in Quark.
  762.     $this->nthChild($groupSize, $elementInGroup, TRUE);
  763.   }
  764.   
  765.   /**
  766.    * Get a list of peer elements.
  767.    * If $requireSameTag is TRUE, then only peer elements with the same
  768.    * tagname as the given element will be returned.
  769.    *
  770.    * @param $element 
  771.    *   A DomElement.
  772.    * @param $requireSameTag 
  773.    *   Boolean flag indicating whether all matches should have the same
  774.    *   element name (tagName) as $element.
  775.    * @return 
  776.    *   Array of peer elements.
  777.    */
  778.   protected function listPeerElements($element, $requireSameTag = FALSE) {
  779.     $peers = array();
  780.     $parent = $element->parentNode;
  781.     foreach ($parent->childNodes as $node) {
  782.       if ($node->nodeType == XML_ELEMENT_NODE) {
  783.         if ($requireSameTag) {
  784.           // Need to make sure that the tag matches:
  785.           if ($element->tagName == $node->tagName) {
  786.             $peers[] = $node;
  787.           }
  788.         }
  789.         else {
  790.           $peers[] = $node;
  791.         }
  792.       }
  793.     }
  794.     return $peers;
  795.   }
  796.   
  797.   /**
  798.    * Get the nth child (by index) from matching candidates.
  799.    *
  800.    * This is used by pseudo-class handlers.
  801.    */
  802.   protected function childAtIndex($index, $tagName = NULL) {
  803.     $restrictToElement = !$this->findAnyElement;
  804.     $matches = $this->candidateList();
  805.     $defaultTagName = $tagName;
  806.     
  807.     // XXX: Added in Quark: I believe this should return an empty
  808.     // match set if no child was found tat the index.
  809.     $this->matches = new SplObjectStorage();
  810.     
  811.     foreach ($matches as $item) {
  812.       $parent = $item->parentNode;
  813.       
  814.       // If a default tag name is supplied, we always use it.
  815.       if (!empty($defaultTagName)) {
  816.         $tagName = $defaultTagName;
  817.       }
  818.       // If we are inside of an element selector, we use the 
  819.       // tag name of the given elements.
  820.       elseif ($restrictToElement) {
  821.         $tagName = $item->tagName;
  822.       }
  823.       // Otherwise, we skip the tag name match.
  824.       else {
  825.         $tagName = NULL;
  826.       }
  827.  
  828.       // Loop through all children looking for matches.
  829.       $i = 0;
  830.       foreach ($parent->childNodes as $child) {
  831.         if ($child->nodeType !== XML_ELEMENT_NODE) {
  832.           break; // Skip non-elements
  833.         }
  834.         
  835.         // If type is set, then we do type comparison
  836.         if (!empty($tagName)) {
  837.           // Check whether tag name matches the type.
  838.           if ($child->tagName == $tagName) {
  839.             // See if this is the index we are looking for.
  840.             if ($i == $index) {
  841.               //$this->matches = new SplObjectStorage();
  842.               $this->matches->attach($child);
  843.               return;
  844.             }
  845.             // If it's not the one we are looking for, increment.
  846.             ++$i;
  847.           }
  848.         }
  849.         // We don't care about type. Any tagName will match.
  850.         else {
  851.           if ($i == $index) {
  852.             $this->matches->attach($child);
  853.             return;
  854.           }
  855.           ++$i;
  856.         }
  857.       } // End foreach
  858.     }
  859.   }

  862.    * Pseudo-class handler for nth-of-type-child.
  863.    * Not implemented.
  864.    */
  865.   protected function nthOfTypeChild($groupSize, $elementInGroup) {
  866.     // EXPERIMENTAL: New in Quark. This should be substantially faster
