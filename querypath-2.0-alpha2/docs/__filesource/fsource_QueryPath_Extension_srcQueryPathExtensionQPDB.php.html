  1. <?php
  2. /**
  3.  * This package contains classes for handling database transactions from
  4.  * within QueryPath.
  5.  *
  6.  * The tools here use the PDO (PHP Data Objects) library to execute database
  7.  * functions.
  8.  *
  9.  * Using tools in this package, you can write QueryPath database queries
  10.  * that query an RDBMS and then insert the results into the document.
  11.  *
  12.  * Example:
  13.  * <code>
  14.  * <?php
  15.  * $template = '<?xml version="1.0"?><tr><td class="colOne"/><td class="colTwo"/><td class="colThree"/></tr>';
  16.  * $qp = qp(QueryPath::HTML_STUB, 'body') // Open a stub HTML doc and select <body/>
  17.  *  ->append('<table><tbody/></table>')
  18.  *  ->dbInit($this->dsn)
  19.  *  ->queryInto('SELECT * FROM qpdb_test WHERE 1', array(), $template)
  20.  *  ->doneWithQuery()
  21.  *  ->writeHTML();
  22.  * ?>
  23.  * </code>
  24.  *
  25.  * The code above will take the results of a SQL query and insert them into a n
  26.  * HTML table.
  27.  *
  28.  * If you are doing many database operations across multiple QueryPath objects,
  29.  * it is better to avoid using {@link QPDB::dbInit()}. Instead, you should
  30.  * call the static {@link QPDB::baseDB()} method to configure a single database
  31.  * connection that can be shared by all {@link QueryPath} instances.
  32.  *
  33.  * Thus, we could rewrite the above to look like this:
  34.  * <code>
  35.   * <?php
  36.   * QPDB::baseDB($someDN);
  37.   *
  38.   * $template = '<?xml version="1.0"?><tr><td class="colOne"/><td class="colTwo"/><td class="colThree"/></tr>';
  39.   * $qp = qp(QueryPath::HTML_STUB, 'body') // Open a stub HTML doc and select <body/>
  40.   *  ->append('<table><tbody/></table>')
  41.   *  ->queryInto('SELECT * FROM qpdb_test WHERE 1', array(), $template)
  42.   *  ->doneWithQuery()
  43.   *  ->writeHTML();
  44.   * ?>
  45.   * </code>
  46.  * Note that in this case, the QueryPath object doesn't need to call a method to
  47.  * activate the database. There is no call to {@link dbInit()}. Instead, it checks
  48.  * the base class to find the shared database.
  49.  *
  50.  * (Note that if you were to add a dbInit() call to the above, it would create
  51.  * a new database connection.)
  52.  *
  53.  * The result of both of these examples will be identical.
  54.  * The output looks something like this:
  55.  * 
  56.  * <code>
  57.  * <!DOCTYPE html PUBLIC "-//W3C//DTD XHTML 1.0 Strict//EN" "http://www.w3.org/TR/xhtml1/DTD/xhtml1-strict.dtd">
  58.  * <html xmlns="http://www.w3.org/1999/xhtml">
  59.  * <head>
  60.  *  <meta http-equiv="Content-Type" content="text/html; charset=utf-8"></meta>
  61.  *     <title>Untitled</title>
  62.  * </head>
  63.  *<body>
  64.  *<table>
  65.  * <tbody>
  66.  *  <tr>
  67.  *   <td class="colOne">Title 0</td>
  68.  *   <td class="colTwo">Body 0</td>
  69.  *   <td class="colThree">Footer 0</td>
  70.  *  </tr>
  71.  *  <tr>
  72.  *   <td class="colOne">Title 1</td>
  73.  *   <td class="colTwo">Body 1</td>
  74.  *   <td class="colThree">Footer 1</td>
  75.  *  </tr>
  76.  *  <tr>
  77.  *   <td class="colOne">Title 2</td>
  78.  *   <td class="colTwo">Body 2</td>
  79.  *   <td class="colThree">Footer 2</td>
  80.  *  </tr>
  81.  *  <tr>
  82.  *   <td class="colOne">Title 3</td>
  83.  *   <td class="colTwo">Body 3</td>
  84.  *   <td class="colThree">Footer 3</td>
  85.  *  </tr>
  86.  *  <tr>
  87.  *   <td class="colOne">Title 4</td>
  88.  *   <td class="colTwo">Body 4</td>
  89.  *   <td class="colThree">Footer 4</td>
  90.  *  </tr>
  91.  * </tbody>
  92.  *</table>
  93.  *</body>
  94.  *</html>
  95.  *</code>

  108. /**
  109.  * Provide DB access to a QueryPath object.
  110.  *
  111.  * This extension provides tools for communicating with a database using the
  112.  * QueryPath library. It relies upon PDO for underlying database communiction. This
  113.  * means that it supports all databases that PDO supports, including MySQL,
  114.  * PostgreSQL, and SQLite.
  115.  *
  116.  * Here is an extended example taken from the unit tests for this library.
  117.  * 
  118.  * Let's say we create a database with code like this:
  119.  * <code>
  120.  *<?php
  121.  * public function setUp() {
  122.  *   $this->db = new PDO($this->dsn);
  123.  *   $this->db->setAttribute(PDO::ATTR_ERRMODE, PDO::ERRMODE_EXCEPTION);
  124.  *   $this->db->exec('CREATE TABLE IF NOT EXISTS qpdb_test (colOne, colTwo, colThree)');
  125.  *   
  126.  *   $stmt = $this->db->prepare(
  127.  *     'INSERT INTO qpdb_test (colOne, colTwo, colThree) VALUES (:one, :two, :three)'
  128.  *   );
  129.  *   
  130.  *   for ($i = 0; $i < 5; ++$i) {
  131.  *     $vals = array(':one' => 'Title ' . $i, ':two' => 'Body ' . $i, ':three' => 'Footer ' . $i);
  132.  *     $stmt->execute($vals);
  133.  *     $stmt->closeCursor();
  134.  *   }
  135.  * }
  136.  * ?>
  137.  * </code>
  138.  * 
  139.  * From QueryPath with QPDB, we can now do very elaborate DB chains like this:
  140.  *<code>
  141.  * <?php
  142.  * $sql = 'SELECT * FROM qpdb_test';
  143.  * $args = array();
  144.  * $qp = qp(QueryPath::HTML_STUB, 'body') // Open a stub HTML doc and select <body/>
  145.  *   ->append('<h1></h1>') // Add <h1/>
  146.  *   ->children()  // Select the <h1/>
  147.  *   ->dbInit($this->dsn) // Connect to the database
  148.  *   ->query($sql, $args) // Execute the SQL query
  149.  *   ->nextRow()  // Select a row. By default, no row is selected.
  150.  *   ->appendColumn('colOne') // Append Row 1, Col 1 (Title 0)
  151.  *   ->parent() // Go back to the <body/>
  152.  *   ->append('<p/>') // Append a <p/> to the body
  153.  *   ->find('p')  // Find the <p/> we just created.
  154.  *   ->nextRow() // Advance to row 2
  155.  *   ->prependColumn('colTwo') // Get row 2, col 2. (Body 1)
  156.  *   ->columnAfter('colThree') // Get row 2 col 3. (Footer 1)
  157.  *   ->doneWithQuery() // Let QueryPath clean up. YOU SHOULD ALWAYS DO THIS.
  158.  *   ->writeHTML(); // Write the output as HTML.
  159.  * ?>
  160.  *</code>
  161.  * With the code above, we step through the document, selectively building elements
  162.  * as we go, and then populating this elements with data from our initial query.
  163.  *
  164.  * When the last command, {@link QueryPath:::writeHTML()}, is run, we will get output
  165.  * like this:
  166.  * <code>
  167.  *   <!DOCTYPE html PUBLIC "-//W3C//DTD XHTML 1.0 Strict//EN" "http://www.w3.org/TR/xhtml1/DTD/xhtml1-strict.dtd">
  168.  *   <html xmlns="http://www.w3.org/1999/xhtml">
  169.  *     <head>
  170.  *         <meta http-equiv="Content-Type" content="text/html; charset=utf-8" />
  171.  *         <title>Untitled</title>
  172.  *     </head>
  173.  *     <body>
  174.  *       <h1>Title 0</h1>
  175.  *       <p>Body 1</p>
  176.  *       Footer 1</body>
  177.  *    </html>
  178.  * </code>
  179.  * Notice the body section in particular. This is where the data has been
  180.  * inserted.
  181.  *
  182.  * Sometimes you want to do something a lot simpler, like give QueryPath a
  183.  * template and have it navigate a query, inserting the data into a template, and
  184.  * then inserting the template into the document. This can be done simply with
  185.  * the {@link queryInto()} function.
  186.  *
  187.  * Here's an example from another unit test:
  188.  * <code>
  189.  * <?php
  190.  * $template = '<?xml version="1.0"?><li class="colOne"/>';
  191.  * $sql = 'SELECT * FROM qpdb_test';
  192.  * $args = array();
  193.  * $qp = qp(QueryPath::HTML_STUB, 'body')
  194.  *   ->append('<ul/>') // Add a new <ul/>
  195.  *   ->children() // Select the <ul/>
  196.  *   ->dbInit($this->dsn) // Initialize the DB
  197.  *   // BIG LINE: Query the results, run them through the template, and insert them.
  198.  *   ->queryInto($sql, $args, $template)
  199.  *   ->doneWithQuery()
  200.  *   ->writeHTML(); // Write the results as HTML.
  201.  * ?>
  202.  * </code>
  203.  * The simple code above puts the first column of the select statement
  204.  * into an unordered list. The example output looks like this:
  205.  * <code>
  206.  * <!DOCTYPE html PUBLIC "-//W3C//DTD XHTML 1.0 Strict//EN" "http://www.w3.org/TR/xhtml1/DTD/xhtml1-strict.dtd">
  207.  * <html xmlns="http://www.w3.org/1999/xhtml">
  208.  *   <head>
  209.  *       <meta http-equiv="Content-Type" content="text/html; charset=utf-8"></meta>
  210.  *       <title>Untitled</title>
  211.  *    </head>
  212.  *    <body>
  213.  *    <ul>
  214.  *    <li class="colOne">Title 0</li>
  215.  *    <li class="colOne">Title 1</li>
  216.  *    <li class="colOne">Title 2</li>
  217.  *    <li class="colOne">Title 3</li>
  218.  *    <li class="colOne">Title 4</li>
  219.  *    </ul>
  220.  *   </body>
  221.  * </html>
  222.  * </code>
  223.  *
  224.  * Typical starting methods for this class are {@link QPDB::baseDB()},
  225.  * {@link QPDB::query()}, and {@link QPDB::queryInto()}.
  226.  */
  227. class QPDB implements QueryPathExtension {
  228.   protected $qp;
  229.   protected $dsn;
  230.   protected $db;
  231.   protected $opts;
  232.   protected $row = NULL;
  233.   protected $stmt = NULL;
  234.   
  235.   protected static $con = NULL;
  236.   
  237.   /**
  238.    * Create a new database instance for all QueryPath objects to share.
  239.    *
  240.    * This method need be called only once. From there, other QPDB instances
  241.    * will (by default) share the same database instance.
  242.    *
  243.    * Normally, a DSN should be passed in. Username, password, and db params
  244.    * are all passed in using the options array.
  245.    *
  246.    * On rare occasions, it may be more fitting to pass in an existing database
  247.    * connection (which must be a {@link PDO} object). In such cases, the $dsn
  248.    * parameter can take a PDO object instead of a DSN string. The standard options
  249.    * will be ignored, though.
  250.    *
  251.    * <b>Warning:</b> If you pass in a PDO object that is configured to NOT throw
  252.    * exceptions, you will need to handle error checking differently.
  253.    *
  254.    * <b>Remember to always use {@link QPDB::doneWithQuery()} when you are done
  255.    * with a query. It gives PDO a chance to clean up open connections that may
  256.    * prevent other instances from accessing or modifying data.</b>
  257.    *
  258.    * @param string $dsn 
  259.    *   The DSN of the database to connect to. You can also pass in a PDO object, which
  260.    *   will set the QPDB object's database to the one passed in.
  261.    * @param array $options 
  262.    *   An array of configuration options. The following options are currently supported:
  263.    *   - username => (string)
  264.    *   - password => (string)
  265.    *   - db params => (array) These will be passed into the new PDO object.
  266.    *     See the PDO documentation for a list of options. By default, the
  267.    *     only flag set is {@link PDO::ATTR_ERRMODE}, which is set to
  268.    *     {@link PDO::ERRMODE_EXCEPTION}.
  269.    * @throws PDOException
  270.    *   An exception may be thrown if the connection cannot be made.
  271.    */
  272.   static function baseDB($dsn, $options = array()) {
  273.     
  274.     $opts = $options + array(
  275.       'username' => NULL,
  276.       'password' => NULL,
  277.       'db params' => array(PDO::ATTR_ERRMODE => PDO::ERRMODE_EXCEPTION),
  278.     );
  279.     
  280.     // Allow this to handle the case where an outside
  281.     // connection does the initialization.
  282.     if ($dsn instanceof PDO) {
  283.       self::$con = $dsn;
  284.       return;
  285.     }
  286.     self::$con = new PDO($dsn, $opts['username'], $opts['password'], $opts['db params']);
  287.   }
  288.   
  289.   /**
  290.    * 
  291.    * This method may be used to share the connection with other,
  292.    * non-QueryPath objects.
  293.    */
  294.   static function getBaseDB() {return self::$con;}
  295.   
  296.   /**
  297.    * Used to control whether or not all rows in a result should be cycled through.
  298.    */
  299.   protected $cycleRows = FALSE;

  301.   /**
  302.    * Construct a new QPDB object. This is usually done by QueryPath itself.
  303.    */
  304.   public function __construct(QueryPath $qp) {
  305.     $this->qp = $qp;
  306.     // By default, we set it up to use the base DB.
  307.     $this->db = self::$con;
  308.   }
  309.   
  310.   /**
  311.    * Create a new connection to the database. Use the PDO DSN syntax for a
  312.    * connection string.
  313.    *
  314.    * This creates a database connection that will last for the duration of
  315.    * the QueryPath object. This method ought to be used only in two cases:
  316.    * - When you will only run a couple of queries during the life of the
  317.    *   process.
  318.    * - When you need to connect to a database that will only be used for
  319.    *   a few things.
  320.    * Otherwise, you should use {@link QPDB::baseDB} to configure a single
  321.    * database connection that all of {@link QueryPath} can share.
  322.    *
  323.    * <b>Remember to always use {@link QPDB::doneWithQuery()} when you are done
  324.    * with a query. It gives PDO a chance to clean up open connections that may
  325.    * prevent other instances from accessing or modifying data.</b>
  326.    *
  327.    * @param string $dsn 
  328.    *   The PDO DSN connection string.
  329.    * @param array $options 
  330.    *   Connection options. The following options are supported:
  331.    *   - username => (string)
  332.    *   - password => (string)
  333.    *   - db params => (array) These will be passed into the new PDO object.
  334.    *     See the PDO documentation for a list of options. By default, the
  335.    *     only flag set is {@link PDO::ATTR_ERRMODE}, which is set to
  336.    *     {@link PDO::ERRMODE_EXCEPTION}.
  337.    * @return QueryPath 
  338.    *   The QueryPath object.
  339.    * @throws PDOException
  340.    *   The PDO library is configured to throw exceptions, so any of the
  341.    *   database functions may throw a PDOException.
  342.    */
  343.   public function dbInit($dsn, $options = array()) {
  344.     $this->opts = $options + array(
  345.       'username' => NULL,
  346.       'password' => NULL,
  347.       'db params' => array(PDO::ATTR_ERRMODE => PDO::ERRMODE_EXCEPTION),
  348.     );
  349.     $this->dsn = $dsn;
  350.     $this->db = new PDO($dsn);
  351.     foreach ($this->opts['db params'] as $key => $val)
  352.       $this->db->setAttribute($key, $val);
  353.     
  354.     return $this->qp;
  355.   }
  356.   
  357.   /**
  358.    * Execute a SQL query, and store the results.
  359.    *
  360.    * This will execute a SQL query (as a prepared statement), and then store
  361.    * the results internally for later use. The data can be iterated using
  362.    * {@link nextRow()}. QueryPath can also be instructed to do internal iteration
  363.    * using the {@link withEachRow()} method. Finally, on the occasion that the
  364.    * statement itself is needed, {@link getStatement()} can be used.
  365.    *
  366.    * Use this when you need to access the results of a query, or when the
  367.    * parameter to a query should be escaped. If the query takes no external
  368.    * parameters and does not return results, you may wish to use the
  369.    * (ever so slightly faster) {@link exec()} function instead.
  370.    * 
  371.    * Make sure you use {@link doneWithQuery()} after finishing with the database
  372.    * results returned by this method.
  373.    *
  374.    * <b>Usage</b>
  375.    * 
  376.    * Here is a simple example:
  377.    * <code>
  378.    * <?php
  379.    * QPQDB::baseDB($someDSN);
  380.    * 
  381.    * $args = array(':something' => 'myColumn');
  382.    * qp()->query('SELECT :something FROM foo', $args)->doneWithQuery();
  383.    * ?>
  384.    * </code>
  385.    *
  386.    * The above would execute the given query, substituting myColumn in place of
  387.    * :something before executing the query The {@link doneWithQuery()} method
  388.    * indicates that we are not going to use the results for anything. This method
  389.    * discards the results.
  390.    *
  391.    * A more typical use of the query() function would involve inserting data
  392.    * using {@link appendColumn()}, {@link prependColumn()}, {@link columnBefore()},
  393.    * or {@link columnAfter()}. See the main documentation for {@link QPDB} to view
  394.    * a more realistic example.
  395.    * 
  396.    * @param string $sql 
  397.    *   The query to be executed.
  398.    * @param array $args 
  399.    *   An associative array of substitutions to make.
  400.    * @throws PDOException
  401.    *   Throws an exception if the query cannot be executed.
  402.    */
  403.   public function query($sql, $args = array()) {
  404.     $this->stmt = $this->db->prepare($sql);
  405.     $this->stmt->execute($args);
  406.     return $this->qp;
  407.   }

  409.   /**
  410.    * Query and append the results.
  411.    *
  412.    * Run a query and inject the results directly into the
  413.    * elements in the QueryPath object.
  414.    *
  415.    * If the third argument is empty, the data will be inserted directly into
  416.    * the QueryPath elements unaltered. However, if a template is provided in
  417.    * the third parameter, the query data will be merged into that template
  418.    * and then be added to each QueryPath element.
  419.    *
  420.    * The template will be merged once for each row, even if no row data is
  421.    * appended into the template.
  422.    *
  423.    * A template is simply a piece of markup labeled for insertion of
  424.    * data. See {@link QPTPL} and {@link QPTPL.php} for more information.
  425.    *
  426.    * Since this does not use a stanard {@link query()}, there is no need
  427.    * to call {@link doneWithQuery()} after this method.
  428.    *
  429.    * @param string $sql 
  430.    *   The SQL query to execute. In this context, the query is typically a
  431.    *   SELECT statement.
  432.    * @param array $args 
  433.    *   An array of arguments to be substituted into the query. See {@link query()}
  434.    *   for details.
  435.    * @param mixed $template 
  436.    *   A template into which query results will be merged prior to being appended
  437.    *   into the QueryPath. For details on the template, see {@link QPTPL::tpl()}.
  438.    * @see QPTPL.php
  439.    * @see QPTPL::tpl()
  440.    * @see query()
  441.    */
  442.   public function queryInto($sql, $args = array(), $template = NULL) {
  443.     $stmt = $this->db->prepare($sql);
  444.     $stmt->setFetchMode(PDO::FETCH_ASSOC);
  445.     $stmt->execute($args);
  446.     
  447.     // If no template, put all values in together.
  448.     if (empty($template)) {
  449.       foreach ($stmt as $row) foreach ($row as $datum) $this->qp->append($datum);
  450.     }
  451.     // Otherwise, we run the results through a template, and then append.
  452.     else {
  453.       foreach ($stmt as $row) $this->qp->tpl($template, $row);
  454.     }
  455.     
  456.     $stmt->closeCursor();
  457.     return $this->qp;
  458.   }

  460.   /**
  461.    * Free up resources when a query is no longer used.
  462.    *
  463.    * This function should <i>always</i> be called when the database
  464.    * results for a query are no longer needed. This frees up the
  465.    * database cursor, discards the data, and resets resources for future
  466.    * use.
  467.    *
  468.    * If this method is not called, some PDO database drivers will not allow
  469.    * subsequent queries, while others will keep tables in a locked state where
  470.    * writes will not be allowed.
  471.    *
  472.    * @return QueryPath 
  473.    *   The QueryPath object.
  474.    */
  475.   public function doneWithQuery() {
  476.     if (isset($this->stmt) && $this->stmt instanceof PDOStatement) {
  477.       // Some drivers choke if results haven't been iterated.
  478.       //while($this->stmt->fetch()) {}
  479.       $this->stmt->closeCursor();
  480.     }
  481.       
  482.     unset($this->stmt);
  483.     $this->row = NULL;
  484.     $this->cycleRows = FALSE;
  485.     return $this->qp;
  486.   }

  489.    * Execute a SQL query, but expect no value.
  490.    * 
  491.    * If your SQL query will have parameters, you are encouraged to
  492.    * use {@link query()}, which includes built-in SQL Injection
  493.    * protection.
  494.    *
  495.    * @param string $sql 
  496.    *   A SQL statement.

  498.    *   An exception will be thrown if a query cannot be executed.
  499.    */

  501.     $this->db->exec($sql);
  502.     return $this->qp;
  503.   }

  505.   /**
  506.    * Advance the query results row cursor.
  507.    *
  508.    * In a result set where more than one row was returned, this will
  509.    * move the pointer to the next row in the set.
  510.    *
  511.    * The PDO library does not have a consistent way of determining how many
  512.    * rows a result set has. The suggested technique is to first execute a
  513.    * COUNT() SQL query and get the data from that.

  526.   /**

  529.    * This is used primarily to instruct QPDB to iterate through all of the
  530.    * rows when appending, prepending, inserting before, or inserting after.

  540.     $this->cycleRows = TRUE;
  541.     return $this->qp;
  542.   }

  544.   /**
  545.    * This is the implementation behind the append/prepend and before/after methods.
  546.    *
  547.    * @param mixed $columnName 
  548.    *   The name of the column whose data should be added to the currently selected
  549.    *   elements. This can be either a string or an array of strings.
  550.    * @param string $qpFunc 
  551.    *   The name of the QueryPath function that should be executed to insert data
  552.    *   into the object.
  553.    * @param string $wrap 
  554.    *   The HTML/XML markup that will be used to wrap around the column data before
  555.    *   the data is inserted into the QueryPath object.
  556.    */
  557.   protected function addData($columnName, $qpFunc = 'append', $wrap = NULL) {
  558.     $columns = is_array($columnName) ? $columnName : array($columnName);
  559.     $hasWrap = !empty($wrap);
  560.     if ($this->cycleRows) {
  561.       while (($row = $this->stmt->fetch(PDO::FETCH_ASSOC)) !== FALSE) {
  562.         foreach ($columns as $col) {
  563.           if (isset($row[$col])) {
  564.             $data = $row[$col];
  565.             if ($hasWrap) 
  566.               $data = qp()->append($wrap)->deepest()->append($data)->top();
  567.             $this->qp->$qpFunc($data);
  568.           }
  569.         }
  570.       }
  571.       $this->cycleRows = FALSE;
  572.       $this->doneWithQuery();
  573.     }
  574.     else {
  575.       if ($this->row !== FALSE) {
  576.         foreach ($columns as $col) {
  577.           if (isset($this->row[$col])) {
  578.             $data = $this->row[$col];
  579.             if ($hasWrap) 
  580.               $data = qp()->append($wrap)->deepest()->append($data)->top();
  581.             $this->qp->$qpFunc($data);
  582.           }
  583.         }
  584.       }
  585.     }
  586.     return $this->qp;
  587.   }
