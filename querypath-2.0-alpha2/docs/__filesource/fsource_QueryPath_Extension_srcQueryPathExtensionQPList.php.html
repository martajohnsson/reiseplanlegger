  66.   /**

  77.    *   Traversable objects can also be used.
  78.    * @param string $type 
  79.    *   One of ul, ol, or dl. Predefined constants are available for use.

  84.   public function appendList($items, $type = self::UL, $options = array()) {
  85.     $opts = $options + array(

  106.   protected function listImpl($items, $type, $opts, $q = NULL) {
  107.     $ele = '<' . $type . '/>';
  108.     if (!isset($q))
  109.       $q = qp()->append($ele)->addClass($opts['list class']);
  110.           
  111.     foreach ($items as $li) {
  112.       if ($li instanceof QueryPath) {
  113.         $q = $this->listImpl($li->get(), $type, $opts, $q);
  114.       }
  115.       elseif (is_array($li) || $li instanceof Traversable) {

  118.         $q->parent();
  119.       }
  120.       else {
  121.         $q->append('<li>' . $li . '</li>');
  122.       }
  123.     }
  124.     return $q;
