  2. /**
  3.  * This file contains the Query Path extension tools.
  4.  *
  5.  * Query Path can be extended to support additional features. To do this,
  6.  * you need only create a new class that implements {@link QueryPathExtension}
  7.  * and add your own methods. This class can then be registered as an extension.
  8.  * It will then be available through Query Path.
  9.  *
  10.  * For information on building your own extension, see {@link QueryPathExtension}.
  11.  * If you are trying to load an extension you have downloaded, chances are good that
  12.  * all you need to do is {@link require_once} the file that contains the extension.
  13.  *
  14.  * @package QueryPath
  15.  * @author M Butcher <matt@aleph-null.tv>
  16.  * @license http://opensource.org/licenses/lgpl-2.1.php LGPL or MIT-like license.
  17.  * @see QueryPathExtension
  18.  * @see QueryPathExtensionRegistry::extend()
  19.  */

  21. /**
  22.  * A QueryPathExtension is a tool that extends the capabilities of a QueryPath object.
  23.  *
  24.  * Extensions to QueryPath should implement the QueryPathExtension interface. The
  25.  * only requirement is that the extension provide a constructor that takes a
  26.  * QueryPath object as a parameter.
  27.  *
  28.  * Here is an example QueryPath extension:
  29.  * <code><?php
  30.  * class StubExtensionOne implements QueryPathExtension {
  31.  *   private $qp = NULL;
  32.  *   public function __construct(QueryPath $qp) {
  33.  *     $this->qp = $qp;
  34.  *   }
  35.  *
  36.  *   public function stubToe() {
  37.  *     $this->qp->find(':root')->append('<toe/>')->end();
  38.  *     return $this->qp;
  39.  *   }
  40.  * }
  41.  * QueryPathExtensionRegistry::extend('StubExtensionOne');
  42.  * ?></code>
  43.  * In this example, the StubExtensionOne class implements QueryPathExtension.
  44.  * The constructor stores a local copyof the QueryPath object. This is important
  45.  * if you are planning on fully integrating with QueryPath's Fluent Interface.
  46.  *
  47.  * Finally, the stubToe() function illustrates how the extension makes use of
  48.  * QueryPath internally, and remains part of the fluent interface by returning
  49.  * the $qp object.
  50.  *
  51.  * Notice that beneath the class, there is a single call to register the
  52.  * extension with QueryPath's registry. Your extension should end with a line
  53.  * similar to this.
  54.  *
  55.  * <b>How is a QueryPath extension called?</b>
  56.  *
  57.  * QueryPath extensions are called like regular QueryPath functions. For
  58.  * example, the extension above can be called like this:
  59.  * <code>
  60.  * qp('some.xml')->stubToe();
  61.  * </code>
  62.  * Since it returns the QueryPath ($qp) object, chaining is supported:
  63.  * <code>
  64.  * print qp('some.xml')->stubToe()->xml();
  65.  * </code>
  66.  * When you write your own extensions, anything that does not need to return a
  67.  * specific value should return the QueryPath object. Between that and the
  68.  * extension registry, this will provide the best developer experience.
  69.  */
  70. interface QueryPathExtension {
  71.   public function __construct(QueryPath $qp);
  72. }

  74. /**
  75.  * A registry for QueryPath extensions.
  76.  *
  77.  * QueryPath extensions should call the {@link QueryPathExtensionRegistry::extend()}
  78.  * function to register their extension classes. The QueryPath library then
  79.  * uses this information to determine what QueryPath extensions should be loaded and
  80.  * executed.
  81.  */
  82. class QueryPathExtensionRegistry {
  83.   /**
  84.    * Internal flag indicating whether or not the registry should
  85.    * be used for automatic extension loading. If this is false, then
  86.    * implementations should not automatically load extensions.
  87.    */
  88.   public static $useRegistry = TRUE;
  89.   /**
  90.    * The extension registry. This should consist of an array of class
  91.    * names.
  92.    */
  93.   protected static $extensionRegistry = array();
  94.   protected static $extensionMethodRegistry = array();
  95.   /**
  96.    * Extend QueryPath with the given extension class.
  97.    */
  98.   public static function extend($classname) {
  99.     self::$extensionRegistry[] = $classname;
  100.     $class = new ReflectionClass($classname);
  101.     $methods = $class->getMethods();
  102.     foreach ($methods as $method) {
  103.       self::$extensionMethodRegistry[$method->getName()] = $classname;
  104.     }
  105.   }
  106.   
  107.   /**
  108.    * Check to see if a method is known.
  109.    * This checks to see if the given method name belongs to one of the
  110.    * registered extensions. If it does, then this will return TRUE.
  111.    *
  112.    * @param string $name 
  113.    *   The name of the method to search for.
  114.    * @return boolean 
  115.    *   TRUE if the method exists, false otherwise.
  116.    */
  117.   public static function hasMethod($name) {
  118.     return isset(self::$extensionMethodRegistry[$name]);
  119.   }
  120.   
  121.   /**
  122.    * Check to see if the given extension class is registered.
  123.    * Given a class name for a {@link QueryPathExtension} class, this
  124.    * will check to see if that class is registered. If so, it will return
  125.    * TRUE.
  126.    * 
  127.    * @param string $name 
  128.    *   The name of the class.
  129.    * @return boolean 
  130.    *   TRUE if the class is registered, FALSE otherwise.
  131.    */
  132.   public static function hasExtension($name) {
  133.     return isset(self::$extensionRegistry[$name]);
  134.   }
  135.   
  136.   /**
  137.    * Get the class that a given method belongs to.
  138.    * Given a method name, this will check all registered extension classes
  139.    * to see if any of them has the named method. If so, this will return
  140.    * the classname.
  141.    *
  142.    * Note that if two extensions are registered that contain the same
  143.    * method name, the last one registred will be the only one recognized.
  144.    *
  145.    * @param string $name 
  146.    *   The name of the method.
  147.    * @return string 
  148.    *   The name of the class.
  149.    */
  150.   public static function getMethodClass($name) {
  151.     return self::$extensionMethodRegistry[$name];
  152.   }
  153.   
  154.   /**
  155.    * Get extensions for the given QueryPath object.
  156.    *
  157.    * Given a {@link QueryPath} object, this will return
  158.    * an associative array of extension names to (new) instances.
  159.    * Generally, this is intended to be used internally.
  160.    *
  161.    * @param QueryPath $qp 
  162.    *   The QueryPath into which the extensions should be registered.
  163.    * @return array 
  164.    *   An associative array of classnames to instances.
  165.    */
  166.   public static function getExtensions(QueryPath $qp) {
  167.     $extInstances = array();
  168.     foreach (self::$extensionRegistry as $ext) {
  169.       $extInstances[$ext] = new $ext($qp);
  170.     }
  171.     return $extInstances;
  172.   }
  173.   
  174.   /**
  175.    * Enable or disable automatic extension loading.
  176.    *
  177.    * If extension autoloading is disabled, then QueryPath will not
  178.    * automatically load all registred extensions when a new QueryPath
  179.    * object is created using {@link qp()}.
  180.    */
  181.   public static function autoloadExtensions($boolean = TRUE) {
  182.     self::$useRegistry = $boolean;
  183.   }
  184. }
